While Loops

  1. What is the output of this block of code? 
    int i=5;
int x=10;

while (i <= 7)
{
    if(x > 0)
    {
        i++;
        x = x - 3;
    }
}

System.out.println("x = " + x);
System.out.println("i = " + i);

  2. What is the output of this block of code? 
    int y = -2;
int i = -4;

while (i > -9)
{
    i--;
    y = y + i;
}

System.out.println("y = " + y);
System.out.println("i = " + i);

  3. What is the output of this block of code? 
    int a = 12;
int j = 0;

while (j < 10)
{
    j = j + 2;
    a = a - 3 + j;
}

System.out.println("a = " + a);
System.out.println("j = " + j);

  4. What is the output of this block of code? 
    int i = 0;
int z = 0;

while(z < 12)
{
	z = z + 3;
	i = i + 3;
	if(i > 4)
	{
		i = i - 2;
	}
	else
	{
		i = i + 1;
	}
}

System.out.println("i = " + i);
System.out.println("z = " + z);

  5. What is the output of this block of code? 
    int x = 1;
int y = -1;
int i = 1;
int j = 4;

while (i < 5)
{
    i = i + 2;
    x = i - 1;
    j = 4;

    while (j < 6)
    {
        j++;
        x = x + 2;
        y = x + 3;
        x = y - 2;
    }
}

System.out.println("x = " + x);
System.out.println("y = " + y);
